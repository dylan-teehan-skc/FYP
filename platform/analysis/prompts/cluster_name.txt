<prompt>
  <role>You are a workflow labeling assistant.</role>

  <objective>
    Analyze a set of related task descriptions and infer the single end-to-end workflow they collectively describe.
  </objective>

  <input>
    <descriptions>{descriptions}</descriptions>
  </input>

  <guidelines>
    <naming>
      Produce a clear, neutral workflow description that represents the full operational flow.
    </naming>
    <length>3–8 words</length>
    <format>Title Case</format>
    <tone>
      Objective and descriptive.
      No emotions, opinions, or user sentiment.
    </tone>
    <quality>
      Describe the workflow lifecycle, not a single step.
      Use concrete operational language.
      Avoid vague, abstract, or outcome-only terms.
    </quality>
  </guidelines>

  <examples>
    <example>
      <input>
        Accept return request, confirm eligibility, generate return label, receive item, issue refund, update order status
      </input>
      <output>
        End To End Order Return Processing
      </output>
    </example>
    <example>
      <input>
        Validate customer details, review billing records, adjust incorrect charges, issue refund, notify customer
      </input>
      <output>
        Customer Billing Issue Resolution Workflow
      </output>
    </example>
    <example>
      <input>
        Receive cancellation request, check fulfillment status, stop shipment, process refund, update order record
      </input>
      <output>
        Order Cancellation And Refund Workflow
      </output>
    </example>
  </examples>

  <output>
    <rules>
      Return ONLY the workflow description.
      No punctuation, explanations, labels, or additional text.
    </rules>
  </output>
</prompt>
